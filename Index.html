// --- UPGRADED GENTLE FEMALE VOICE ENGINE --- let systemVoices = []; // Pre-load voices for Chrome & Safari compatibility function loadVoices() { if (typeof window !== 'undefined' && window.speechSynthesis) { systemVoices = window.speechSynthesis.getVoices(); } } if (typeof window !== 'undefined' && window.speechSynthesis) { window.speechSynthesis.onvoiceschanged = loadVoices; loadVoices(); } function speak(text) { if (!appProgress.audioEnabled) return; try { window.speechSynthesis.cancel(); // Stop any overlapping speech immediately const sentence = new SpeechSynthesisUtterance(text); sentence.lang = 'en-US'; // Dynamic search for friendly, warm female voices const femaleVoiceKeywords = ["samantha", "google us english", "zira", "hazel", "karen", "female", "en-us"]; let selectedVoice = null; // Scan device's voice database for the best match if (systemVoices.length === 0) { systemVoices = window.speechSynthesis.getVoices(); } for (let keyword of femaleVoiceKeywords) { selectedVoice = systemVoices.find(voice => voice.name.toLowerCase().includes(keyword) && (voice.lang.startsWith('en') || voice.lang.startsWith('en-US')) ); if (selectedVoice) break; } if (selectedVoice) { sentence.voice = selectedVoice; } // Parent/Teacher adjustments for a soft, friendly delivery: sentence.pitch = 1.35; // Slightly higher pitch for a warm, bright tone sentence.rate = 0.82; // Soft, clear, and patient pacing window.speechSynthesis.speak(sentence); } catch (e) { console.warn("Synthesis engine offline:", e); } }
