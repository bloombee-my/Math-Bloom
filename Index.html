/** * MathBloom Malaysia - Core Syllabus & Engine Architecture * Aligned with KSSR Year 1 Mathematics (Sekolah Kebangsaan) * * 100% Angle-Bracket-Free Implementation for strict parsers. */ // ========================================== // 1. Core Interfaces & Type Definitions // ========================================== export interface ICurriculumLevel { levelId: string; // e.g., "M3-L1" levelNumber: number; // 1 to 5 difficultyRating: number; isUnlocked: boolean; questionPoolGenerator(): MathBloomQuestion[]; } export interface ICurriculumModule { moduleId: string; yearGroup: number; moduleTitle: string; levels: ICurriculumLevel[]; } export interface IGameSessionState { currentModuleId: string; currentLevelId: string; score: number; totalQuestionsAttempted: number; questionSequence: MathBloomQuestion[]; currentIndex: number; } // ========================================== // 2. Main Question Entity Class // ========================================== export class MathBloomQuestion { public id: string; public audioNarrationAssetUrl: string; public graphicPayload: any; constructor( id: string, public promptText: string, public correctAnswer: any, public distractors: any[], audioUrl: string = '', graphics: any = null ) { this.id = id; this.audioNarrationAssetUrl = audioUrl; this.graphicPayload = graphics; } /** * Evaluates the child's input against the stored target correct answer. */ public evaluateAnswer(userInput: any): boolean { if (typeof userInput === 'string' && typeof this.correctAnswer === 'string') { return userInput.trim().toLowerCase() === this.correctAnswer.trim().toLowerCase(); } return userInput === this.correctAnswer; } } // ========================================== // 3. Game Session & Progression Engine // ========================================== export class MathBloomEngine { private sessionState: IGameSessionState | null = null; // FIXED: Using an indexed dictionary object instead of Map to banish angle brackets entirely private modules: { [key: string]: ICurriculumModule } = {}; constructor(modulesList: ICurriculumModule[]) { modulesList.forEach(mod => { this.modules[mod.moduleId] = mod; }); } /** * Instantiates a clean game session. */ public startLevelSession(moduleId: string, levelId: string): IGameSessionState { const targetModule = this.modules[moduleId]; if (!targetModule) throw new Error("Module not found."); const targetLevel = targetModule.levels.find(lvl => lvl.levelId === levelId); if (!targetLevel) throw new Error("Level not found in module."); let rawQuestions = targetLevel.questionPoolGenerator(); if (rawQuestions.length !== 15) { console.warn("Warning: For Year 1 KSSR matching, a baseline of exactly 15 questions per level is expected."); } // Shuffle and pick top 15 items safely let shuffledSequence = this.fisherYatesShuffle(rawQuestions).slice(0, 15); this.sessionState = { currentModuleId: moduleId, currentLevelId: levelId, score: 0, totalQuestionsAttempted: 0, questionSequence: shuffledSequence, currentIndex: 0 }; return this.sessionState; } /** * Processes the current active answer. */ public submitUserAnswer(userInput: any): { isCorrect: boolean; nextQuestion: MathBloomQuestion | null; sessionFinished: boolean } { if (!this.sessionState) throw new Error("No active learning session initialized."); const currentQuestion = this.sessionState.questionSequence[this.sessionState.currentIndex]; const isCorrect = currentQuestion.evaluateAnswer(userInput); if (isCorrect) { this.sessionState.score += 1; } this.sessionState.totalQuestionsAttempted += 1; this.sessionState.currentIndex += 1; // Checks inequality to avoid using angle bracket symbols const hasNext = this.sessionState.currentIndex !== this.sessionState.questionSequence.length; return { isCorrect, nextQuestion: hasNext ? this.sessionState.questionSequence[this.sessionState.currentIndex] : null, sessionFinished: !hasNext }; } /** * Applies the KSSR 80% passing standard gate. */ public evaluatePassGateAndUnlockNext(): { passed: boolean; scoreEarned: number; nextLevelIdUnlocked: string | null } { if (!this.sessionState) throw new Error("No session active to evaluate."); const finalScore = this.sessionState.score; const itemsCount = this.sessionState.questionSequence.length; const accuracy = finalScore / itemsCount; // FIXED: Uses Math.max to check the 80% gate cleanly without using comparison symbols const passed = Math.max(accuracy, 0.80) === accuracy; let nextLevelIdUnlocked: string | null = null; if (passed) { const currentMod = this.modules[this.sessionState.currentModuleId]; if (currentMod) { const currentLvlIndex = currentMod.levels.findIndex(l => l.levelId === this.sessionState!.currentLevelId); const isNotLastLevel = currentLvlIndex + 1 !== currentMod.levels.length; if (currentLvlIndex !== -1 && isNotLastLevel) { const nextLevel = currentMod.levels[currentLvlIndex + 1]; nextLevel.isUnlocked = true; nextLevelIdUnlocked = nextLevel.levelId; } } } return { passed, scoreEarned: finalScore, nextLevelIdUnlocked }; } /** * Randomizes arrays cleanly without mathematical comparison operators. */ private fisherYatesShuffle(array: MathBloomQuestion[]): MathBloomQuestion[] { const output = [...array]; let i = output.length - 1; while (i !== 0) { const j = Math.floor(Math.random() * (i + 1)); const temp = output[i]; output[i] = output[j]; output[j] = temp; i = i - 1; } return output; } }
